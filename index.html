Installation

// 1. Include the CSS and JavaScript files
<link rel="stylesheet" href="css/modal.css">
<script src="js/modal.js"></script>

// 2. Initialize the Modal library
document.addEventListener('DOMContentLoaded', function() {
    Modal.init();
});
